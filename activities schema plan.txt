TABLE divisions

id VARCHAR(10) KEY เช่น MCCC, MUAN, 
type VARCHAR(20) เช่น dept, club
name VARCHAR(100) เช่น ฝ่ายพัฒนาเทคโนโลยีและสารสนเทศ
vice_president

TABLE projects
project_id VARCHAR(36) KEY เช่น MCCC1, MCCC2, MUAN1, MUAN2
owner_id INT(9) รหัสนศ.
division VARCHAR(10) === ASSOCIATE กับ id ของ divisions
name VARCHAR(200) ชื่อโครงการ
breif_info VARCHAR(500) ข้อมูลโครงการแบบบรีฟ
breif_full VARCHAR(10000) ข้อมูลโครงการแบบเต็ม ๆ เป็น markdown
allow_register TINYINT(1) TRUE, FALSE อนุญาตให้สมัคร
register_date_from DATETIME(3) รับสมัครตั้งแต่วันที่
register_date_until DATETIMT(3) รับสมัครถึงวันที่
participant INT(3) จำนวนที่รับ
updatedDateTime วันที่แก้ไขข้อมูล

TABLE project_responsible_person
project_id รหัสโครงการ === ASSOCIATE WITH projects.project_id
student_id รหัสนักศึกษา

TABLE PARTICIPANT
student_id รหัสประจำตัวนักศึกษา
project_id 

TABLE question_list
project_id VARCHAR รหัสประจำโครงการ
question_id INT(2) รหัสคำถาม 01,02,03,04 ไปเรื่อย ๆ
question_text VARCHAR
type VARCHAR

TABLE answers
student_id INT(9)
project_id VARCHAR
question_id VARCHAR
answer (VARCHAR)
